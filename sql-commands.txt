Create database name "mydb"
createdb mydb

List databases 
psql --list

Delete databased named "mydb"
dropdb mydb

See all indexes on table "fields"
SELECT                                                                   
    indexname,                                                                                 
    indexdef
FROM
    pg_indexes
WHERE
    tablename = 'fields';

Restart postgres after a hard reboot
  brew services restart postgresql
